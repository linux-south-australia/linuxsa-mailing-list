Like this?

EXTIF="eth0"
EXTIP=`/sbin/ifconfig | grep -A 4 $EXTIF_0 | awk '/inet/ { print $2 } ' |
sed -e s/addr://`
EXTBCAST=`/sbin/ifconfig | grep -A 1 $EXTIF_0 | awk '/Bcast/ { print $3 } '
| sed -e s/Bcast://`
EXTGW=`/sbin/route -n | grep -A 4 UG | awk '{ print $2 } '`
echo External IP:$EXTIP  -- Broadcast:$EXTBCAST  -- Default Gateway:$EXTGW
