Well it seems some virus checkers are more useful than others. I *had*
posted  the  decoded  virus,  just  for  curiosity, but it bounced, so
rather  than that, here's some quick and dirty C code to decode it for
the  curious out there. (Yes, I was bored and felt the need to code in
C)

#include <stdio.h>

char decode_char(char code);

int main(int argc, char **argv)
{
  char ch = ' ';

  while (ch != '\n') {
    ch = getchar();
    /* Now we start the decode and print the results */
    printf("%c", decode_char(ch));
  }

  return 0;
}

char decode_char(char code)
{
  char newchar = ' ';

  switch ((int)code) {
    case 15: newchar = '\n';
             break;
    case 16: newchar = '\r';
             break;
    case 17: newchar = ' ';
             break;
    case 18: newchar = '\t';
             break;
    default: newchar = code - 2;
             break;
  }

  return newchar;
}
