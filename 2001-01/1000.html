you could try something like this i spose.

#!/usr/bin/perl -w

open(NAMES, "file") || die "can't open file\n";
@list = <NAMES>;
close(NAMES);

open(OUT, ">>output");
foreach $name (@list)
{
        chomp($name);
        ($first, $last) = split(/\,\s/, $name);

        $a = lc(substr($first, 0, 1));
        $b = lc(substr($last, 0, 4));

        $username = "$b$a"."01";

        print OUT "$username\n";
}
close(OUT);
