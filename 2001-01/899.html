try something like

#!/usr/bin/perl -w

open (HANDLE, "names.txt") or die ("error: $!\n");
@lines = <HANDLE>;
close(HANDLE);

open (HANDLE, ">newfile.txt") or die ("error: $!");
foreach $temp (@lines) {
        ($last, $first) = split (/, /, $temp);
        $username = lc(substr($last, 0,3)) . lc(substr($first, 0, 1)) . "01\n";
        print HANDLE $username;
}
close(HANDLE);
exit(0);
