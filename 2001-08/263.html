I beleive that the /etc/xinetd.d/ folder has the configuration files to
be edited for what services are available;


[root@firestation /root]# cd /etc/xinetd.d/
[root@firestation xinetd.d]# ll

[root@firestation xinetd.d]#

then you edit the relevant file say, wu-ftp which looks like this;

# default: on
# description: The wu-ftpd FTP server serves FTP connections. It uses \
#       normal, unencrypted usernames and passwords for authentication.
service ftp
{
        socket_type             = stream
        wait                    = no
        user                    = root
        server                  = /usr/sbin/in.ftpd
        server_args             = -l -a
        log_on_success          += DURATION USERID
        log_on_failure          += USERID
        nice                    = 10
        disable                 = yes
}

there is probably more to it though! so RTFM!! :)
