Fairly simple. Create an alias in /etc/aliases that points to both 
users, and then tell fetchmail to deliver to that alias.

[root@host root]# echo "both:  lou,waveone" >> /etc/aliases
[root@host root]# newaliases
[root@host root]# cat .fetchmailrc

set logfile "/var/log/fetchmail"
set bouncemail
set no spambounce
set properties ""
set daemon 120
poll mail.bigpond.com with proto POP3
  user 'itemsofdistinction' there with password '**********' is 'both' here
