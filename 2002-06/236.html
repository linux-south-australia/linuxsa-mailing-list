Re: Root Kit detection tool required

Something like this should show you whether you have been 0wn3d by a 12
year old:

#include <sys/types.h>

#include <errno.h>
#include <signal.h>

#define PID_MAX	99999	/* 4.4BSD */

int
main(argc, argv)
    int    argc;
    char **argv;
{
	int i;
        extern int errno;

	for (i = 1; i < PID_MAX + 1; i++) {
                errno = 0;
		if(kill(i, 0) != -1 || errno != ESRCH)
			printf("%d\n", i);
        }

	exit(0);
}

Compare to the output of ps.
