/etc/hosts.allow:
Nothing other then comments

/etc/hosts.deny:
Nothing other then comments

/etc/xinet.d/wu-ftpd:
service ftp
{
        socket_type             = stream
        wait                    = no
        user                    = root
        server                  = /usr/sbin/in.ftpd
        server_args             = -l -a
        log_on_success          += DURATION
        nice                    = 10
        disable                 = no
}
