Not sure if read() is what your looking for but here is a small snip
of code using getch from ncurses.

If you want to compile this you will need to link in the ncurses
library  -lncurses

======== Beginning of not very well tested code ======

#include <stdio.h>
#include <curses.h>

int menu(void);

int main(void)
{
  int option;

  option = menu();
  printf("\nThe option entered was %c\n", option);
  return(0);
}


int menu(void)
{
  int lines = 2, cols = 20;
  int index, ch;
  char *options[]={
		"Append to data file",
		"Find a record",
		"List a record"
		};


  initscr();
  cbreak(); /*  Don't forget this one john */
  
  for(index = 0; index < (sizeof(options) / sizeof(char *)); index ++)
  {  
     move( lines++,cols);
     addstr(options[index]);
  }     

  move(lines,cols);
  ch=getch();
  refresh();
  endwin();
  return(ch);
}
