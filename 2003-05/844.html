hello guys 

im a bit new to linux world.  im trying to make my linux box (RHL 8) as
my PDC using Samba .  My client is using windows xp and i want it to get
resources from the PDC.  the follwing is the content of my smb.conf

----------------------------- My smb.conf file contents ----------------

[global]

        workgroup = WORKGROUP
        netbios name = LinuxServ
        encrypt passwords = yes
        update encrypted = yes
        log level = 3
        log file = /var/log/samba/log.%m
        keepalive = 10000
        socket options = TCP_NODELAY SO_SNDBUF=8192 SO_RCVBUF=8192
#       the next directive suggest that all people in the group 'admin'
are domain admins
        domain admin group = @admin
        add user script = /usr/sbin/useradd -d /dev/null -g 200 -s
/bin/false -M %u
        logon script = %u.bat
        logon path = \\%N\profiles\%u
        domain logons = yes
        os level = 33
        preferred master = true
        domain master = true
        wins support = yes
        hide unreadable = yes

[netlogon]
        comment = Netlogon Scripts
        path = /netlogon

[profiles]
        path = /profiles
        read only = No
        create mask = 0600
        directory mask = 0700

[homes]
        comment = Home Directories
        path = /home/%u
        create mask = 0700
        directory mask = 0700
        browseable = No
