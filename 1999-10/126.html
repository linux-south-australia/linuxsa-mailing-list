the following script should work (just remember to keep things quoted to 
prevent things splitting it up) :

#!/bin/ksh

# set IFS (internal field separator) to newline
IFS="
"

for i in $(ls *.mp3)
do
  mpg123 "$i" -w "$(echo $i | tr ".mp3" ".wav")"
done
