Here is the trick I use to find which RPM something is in when the RPM is
not installed:

cd /mnt/cdrom/$dirsto/RPMS
for f in *.rpm
do
   if [ "`rpm -qlp $f | grep my-prog`" != "" ]; then echo $f; fi
done
