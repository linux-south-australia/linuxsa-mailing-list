And the generic 'template' i use for adding new vhosts is:
<VirtualHost host.tld>
    ServerAdmin webmaster@host.tld
    DocumentRoot /home/host/public_html
    ServerName host.tld
    ServerAlias host.tld *.host.tld

    ErrorLog  /home/host/logs/error_log
    CustomLog /home/host/logs/access_log combined

    <Directory />
        Options All
        AllowOverride All
    </Directory>
</VirtualHost>
