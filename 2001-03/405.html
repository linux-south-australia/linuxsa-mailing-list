#! /usr/bin/perl

$subject=`formail -x subject`;

print "on topic" unless $subject=~/
                 #is 
                 OFF TOPIC/m;

if (($subject=~/
 #is
 OFF TOPIC/m;) and(exists $TELSTRA_ADSL{more_appropriate_forum}) ) {
	use more_appropriate_forum;
} else {
 open(MAIL, "|/usr/sbin/sendmail -t");
 select MAIL;
 print "To: g@netcraft.com\n";
 print "From: [insert_your_name]@[insert_your_provider]"
 print "Subject: Please Create the \"LinuxSA Telstra Discussion
List\"\n";
 print "Thanks!\n";
}

if (exists $TELSTRA_ADSL{is_crap}) {
 SWITCH:
   $away_from{telstra}=$now;
}

DSL

(if the subject line does not contain "OFF TOPIC", print "on topic". If
the subject if "OFF TOPIC" and there exists a more appropriate forum to
discuss telstra adsl then use the more appropriate forum. If there isn't
[ a more appropriate forum] send an e-mail to the LINUX-SA organiser
asking him to create a LinuxSA Telstra Discussion List. If "is_crap" is
one of the attributes of TELSTRA_ADSL then switch away_from telstra -
now)
