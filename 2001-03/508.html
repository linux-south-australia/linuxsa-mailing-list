this should take care of what your after:

open(MAIL, "|/usr/sbin/sendmail $notify");
print MAIL "Subject: BAMBE Notification\n";
print MAIL "\n";
print MAIL "<content of email>\n";
print MAIL "\n";
close(MAIL);
